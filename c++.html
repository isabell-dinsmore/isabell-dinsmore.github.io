This was a coding practice we did in my C++ class a few years back. The purpose of the program is to take an input, echo that input out in a sentence, and then take an answer input and print based off what the answer input was.

				
#include < iostream>
#include < string>

using namespace std;

int main()
{
	string ans, name;
	int hold;

	cout << "Welcome to class!" << endl;
	cout << "What is your name?" << endl;
	cin >> name;
	cout << name + "," + " " + "Do you want to learn to code?" << endl;
	cin >> ans;
	if(ans == "yes")
	{
		cout << "Okay, let's learn!" << endl;
	}else if(ans == "no")
	{
		cout << "Okay, goodbye." << endl;
	}else
	{
		cout << "Please enter 'yes' or 'no'." << endl;
	}
	cin >> hold;
	return 0;
}
