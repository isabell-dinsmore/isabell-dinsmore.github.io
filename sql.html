This is SQL code that would make a database that contains customers, products, shops, sales, and vendors. These tables contain primary keys that connect them as well.

				
					-- ****************** SqlDBM: MySQL ******************;
					-- ***************************************************;

					-- ************************************** `customers`

					CREATE TABLE `customers`
					(
					 `phoneNum`    int NOT NULL ,
					 `customerNum` int NOT NULL ,
					 `name`        varchar(45) NOT NULL ,
					 `address`     varchar(45) NOT NULL ,
					PRIMARY KEY (`phoneNum`)
					);

					-- ************************************** `products`

					CREATE TABLE `products`
					(
					 `price`        float NOT NULL ,
					 `type`         varchar(45) NOT NULL ,
					 `manufacturer` varchar(45) NOT NULL ,
					 `address`      varchar(45) NOT NULL ,
					PRIMARY KEY (`price`),
					KEY `fkIdx_38` (`address`),
					CONSTRAINT `FK_38` FOREIGN KEY `fkIdx_38` (`address`) REFERENCES `shops` (`address`)
					);

					-- ************************************** `shops`

					CREATE TABLE `shops`
					(
					 `address` varchar(45) NOT NULL ,
					 `name`    varchar(45) NOT NULL ,
					 `price`   float NOT NULL ,
					PRIMARY KEY (`address`),
					KEY `fkIdx_41` (`price`),
					CONSTRAINT `FK_41` FOREIGN KEY `fkIdx_41` (`price`) REFERENCES `products` (`price`)
					);

					-- ************************************** `sales`

					CREATE TABLE `sales`
					(
					 `products` varchar(45) NOT NULL ,
					 `price`    float NOT NULL ,
					 `address`  varchar(45) NOT NULL ,
					 `staffNum` int NOT NULL ,
					 `phoneNum` int NOT NULL ,
					 `date`     date NOT NULL ,
					 `sumTotal` float NOT NULL ,
					PRIMARY KEY (`products`),
					KEY `fkIdx_22` (`price`),
					CONSTRAINT `FK_22` FOREIGN KEY `fkIdx_22` (`price`) REFERENCES `products` (`price`),
					KEY `fkIdx_25` (`address`),
					CONSTRAINT `FK_25` FOREIGN KEY `fkIdx_25` (`address`) REFERENCES `shops` (`address`),
					KEY `fkIdx_28` (`staffNum`),
					CONSTRAINT `FK_28` FOREIGN KEY `fkIdx_28` (`staffNum`) REFERENCES `vendors` (`staffNum`),
					KEY `fkIdx_31` (`phoneNum`),
					CONSTRAINT `FK_31` FOREIGN KEY `fkIdx_31` (`phoneNum`) REFERENCES `customers` (`phoneNum`)
					);

					-- ************************************** `vendors`

					CREATE TABLE `vendors`
					(
					 `staffNum` int NOT NULL ,
					 `name`     varchar(45) NOT NULL ,
					PRIMARY KEY (`staffNum`)
					);
